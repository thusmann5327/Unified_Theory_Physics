    Z_A = set(addr_A['zeckendorf'])
    Z_B = set(addr_B['zeckendorf'])
    shared = len(Z_A & Z_B)
    total = max(len(Z_A), len(Z_B))
    entanglement = shared / total if total > 0 else 0
    # Higher entanglement = shorter effective transit
    return 1.0 - entanglement
